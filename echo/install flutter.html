# 安装flutter

# 获取flutter git（Get the Flutter SDK）

将其放置在想放的位置

# 配置path

vim ~/.zshrc 或 open ~/.zshrc

export PUB_HOSTED_URL=https://pub.flutter-io.cn  # 国内用户需要设置
export FLUTTER_STORAGE_BASE_URL=https://storage.flutter-io.cn # 国内用户需要设置
export PATH=~/flutter/bin:$PATH #[PATH_TO_FLUTTER_GIT_DIRECTORY]:$PATH

如我把我的flutter_git文件夹放置在Users/XXX/flutter
cat $HOME 输出 Users/xxx

则可以配置为

export PATH="$PATH:$HOME/flutter/bin"

让配置生效

source ~/.zshrc #if you config .zshrc

安装完查看是否安装成功

flutter --version

# Run flutter doctor

flutter doctor
